// Quick Start

Three Lines to Clinical Intelligence

analyze_patient.py
from src.engines import MediScanEngine
from src.utils import AIClient

# Initialize with MiMo API
client = AIClient(api_key="your-key")
engine = MediScanEngine(client)

# Analyze a patient record
report = await engine.analyze(
    document="patient_record.pdf",
    context={"department": "cardiology"}
)

# Results
print(f"Findings: {report['findings']}")
print(f"Critical: {report['critical']}")
print(f"Tokens: {report['tokens_used']}")
